    src/window/window.js

    'use strict'

import { WindowMetadataPrefix } from '../config/constant.js'
import Log from '../logger/default.js'

export function MakeWindow (storage) {
	return {
		_storage: storage,

		// Loaded from storage on window open,
		// then updated in memory
		_bufferSize: 0,

		// In case of persistent storages
		// this cache allow to store
		// metadata once in each event
		// processing
		_cacheMetadata: null,

		// You must call this method each time you change a key
		loadMetadata: async function (key) {
			this._cacheMetadata = await this._storage.get(this.metadataKey(key))
		},

		buffer: async function (key) {
			return this._cacheMetadata == null ? [] : await this._storage.getList(key)
		},

		bufferSize: async function (key) {
			return this._bufferSize	
		},		

		push: async function (key, value) {
			await this._storage.push(key, value)
			//await this._storage.set('kafka-offset.' + value.topic, {offset: value.offset, partition: value.partition, topic: value.topic})
			this._bufferSize += 1
		},

		isOpen: async function (key) {
			return this._cacheMetadata == null ? false : this._cacheMetadata.open
		},	

		hasBeenClosed: async function (key) {
			return this._cacheMetadata == null ? false : this._cacheMetadata.endTime !== null && this._cacheMetadata.open == false
		},			

		startTime: async function (key) {
			return this._cacheMetadata == null ? null : this._cacheMetadata.startTime
		},	

		open: async function (key, startTime) {
			const elements = (await this._storage.getList(key))
			this._bufferSize = elements !== null ? elements.length : 0
			const metadata = {
				open: true,
				startTime: startTime,
				endTime: null		
			}
			this._cacheMetadata = metadata
			await this._storage.set(this.metadataKey(key), metadata)
			Log('info', ['Window open', key, metadata.startTime, metadata.endTime])
		},

		close: async function (key, endTime) {
			let metadata = await this._storage.get(this.metadataKey(key))
			metadata.open = false
			metadata.endTime = endTime
			this._cacheMetadata = metadata

			const windowData = await this._storage.getList(key)
			if (windowData !== null) {
				windowData.map((wd) => {
					wd._windowKey = key 
					wd._starTime = metadata.startTime
					wd._endTime = metadata.endTime 
				})
			}
			Log('info', ['Window close', key, metadata.startTime, metadata.endTime])
			this.onClose(key, windowData)
			this._storage.flush(key)
			this._storage.set(this.metadataKey(key), metadata)
		},

		onClose: function (self) { console.log('to override') },

		metadataKey: function (key) {
			return WindowMetadataPrefix + key
		}
	}
}
